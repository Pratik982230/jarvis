import re

# ---------- 1Ô∏è‚É£ READ FILE ----------
file_name = input("Enter file name (e.g., sample.txt): ")

try:
    with open(file_name, "r") as file:
        content = file.read()
    print("\nüìÑ Original File Content:\n")
    print(content)
except FileNotFoundError:
    print("‚ùå File not found! Please check the file name.")
    exit()

# ---------- 2Ô∏è‚É£ TAKE USER INPUT FOR SEARCH AND REPLACE ----------
search_word = input("\nEnter word/pattern to search: ")
replace_word = input("Enter replacement word: ")

# ---------- 3Ô∏è‚É£ PERFORM SEARCH & REPLACE USING REGEX ----------
new_content = re.sub(search_word, replace_word, content)

# ---------- 4Ô∏è‚É£ DISPLAY UPDATED CONTENT ----------
print("\n‚úÖ Updated File Content:\n")
print(new_content)

# ---------- 5Ô∏è‚É£ SAVE UPDATED CONTENT TO SAME FILE ----------
with open(file_name, "w") as file:
    file.write(new_content)

print("\nüíæ Changes saved successfully to file!")
