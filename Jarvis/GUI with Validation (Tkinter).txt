import tkinter as tk
from tkinter import messagebox

# ---------- Function to Handle Submission ----------
def submit_form():
    name = name_entry.get().strip()
    age = age_entry.get().strip()
    email = email_entry.get().strip()
    gender = gender_var.get()

    # ---- Validations ----
    if not name or not age or not email:
        messagebox.showerror("Error", "âš ï¸ All fields are required!")
        return
    
    if not age.isdigit() or int(age) <= 0:
        messagebox.showerror("Error", "âš ï¸ Age must be a positive number!")
        return
    
    if "@" not in email or "." not in email:
        messagebox.showerror("Error", "âš ï¸ Invalid email format!")
        return
    
    # ---- Display the Data ----
    result_text.delete("1.0", tk.END)  # Clear previous text
    result_text.insert(tk.END, f"âœ… Submitted Successfully!\n\n")
    result_text.insert(tk.END, f"Name: {name}\n")
    result_text.insert(tk.END, f"Age: {age}\n")
    result_text.insert(tk.END, f"Email: {email}\n")
    result_text.insert(tk.END, f"Gender: {gender}\n")

    # Optional success message
    messagebox.showinfo("Success", "Data submitted successfully!")

# ---------- Create Main Window ----------
root = tk.Tk()
root.title("User Registration Form")
root.geometry("400x450")
root.config(bg="#f5f5f5")

# ---------- Labels and Entry Fields ----------
tk.Label(root, text="User Registration Form", font=("Arial", 16, "bold"), bg="#f5f5f5").pack(pady=10)

tk.Label(root, text="Full Name:", bg="#f5f5f5").pack(anchor="w", padx=40)
name_entry = tk.Entry(root, width=30)
name_entry.pack()

tk.Label(root, text="Age:", bg="#f5f5f5").pack(anchor="w", padx=40, pady=(10, 0))
age_entry = tk.Entry(root, width=30)
age_entry.pack()

tk.Label(root, text="Email:", bg="#f5f5f5").pack(anchor="w", padx=40, pady=(10, 0))
email_entry = tk.Entry(root, width=30)
email_entry.pack()

# ---------- Gender Radio Buttons ----------
tk.Label(root, text="Gender:", bg="#f5f5f5").pack(anchor="w", padx=40, pady=(10, 0))
gender_var = tk.StringVar(value="Not Selected")
tk.Radiobutton(root, text="Male", variable=gender_var, value="Male", bg="#f5f5f5").pack(anchor="w", padx=60)
tk.Radiobutton(root, text="Female", variable=gender_var, value="Female", bg="#f5f5f5").pack(anchor="w", padx=60)

# ---------- Submit Button ----------
tk.Button(root, text="Submit", command=submit_form, bg="#4CAF50", fg="white", width=15).pack(pady=15)

# ---------- Text Box to Display Results ----------
tk.Label(root, text="Submitted Data:", bg="#f5f5f5").pack(anchor="w", padx=40)
result_text = tk.Text(root, height=6, width=40)
result_text.pack(pady=5)

# ---------- Run the Window ----------
root.mainloop()
