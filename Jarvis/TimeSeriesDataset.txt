import pandas as pd
import matplotlib.pyplot as plt

# ------------------------------------------------------------
# 1Ô∏è‚É£ LOAD INBUILT TIME SERIES DATASET
# ------------------------------------------------------------
# Pandas has a built-in dataset 'AirPassengers' (available via seaborn or statsmodels)
# Let's use statsmodels for this example
from statsmodels.datasets import get_rdataset

# Load dataset
data = get_rdataset('AirPassengers').data

# Convert to pandas DataFrame with proper DateTime index
data['Month'] = pd.to_datetime(data['time'], format='%Y')
data.set_index('Month', inplace=True)

# ------------------------------------------------------------
# 2Ô∏è‚É£ VIEW DATA
# ------------------------------------------------------------
print("=== üóì Sample Time Series Data ===")
print(data.head())

# ------------------------------------------------------------
# 3Ô∏è‚É£ PLOT THE TIME SERIES
# ------------------------------------------------------------
plt.figure(figsize=(10, 5))
plt.plot(data.index, data['value'], color='blue', marker='o', linestyle='-')
plt.title('‚úàÔ∏è AirPassengers Time Series')
plt.xlabel('Year')
plt.ylabel('Number of Passengers')
plt.grid(True)
plt.show()
